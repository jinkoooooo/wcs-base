Framework operators
```sql
  "eq": =
  "noteq": <>
  "notin": not in
  "is_null": is null
  "is_not_null": is not null
  "is_false": is false
  "is_true": is true
  "is_not_false": is not false
  "is_not_true": is not true
  "is_present": is present
  "is_blank": is blank
  "is_empty_num_id": is empty numeric id
  "gt": >
  "gte": >=
  "lt": <
  "lte": <=
  "nlike": not like
  "between" : "column >= 'value[0]' and column <= 'value[1]'"
    --> ex) between : "a,b"
            column >= 'a' and column <= 'b'
```
